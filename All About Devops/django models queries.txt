--------------AND OR NOT-----------------------
from django.db.models import Q

std.filter(q(department=1) | q( s_age=22)).values('department','s_age')   //OR operator
std.filter(Q(department=1) & Q( s_age=22)).values('department','s_age')   //AND operator
std.filter(~Q(s_name__startswith='a'))		// ~Not operator
std.filter((Q(department=1) | Q( s_age=22)) & Q(s_name__startswith='a')).values_list('s_name','department','s_age') 	//combined logical operators

--------------Annotate Function
Its used to apply some functions on multiple columns

    from django.db.models import *
std.annotate(Count('s_age'))  	//it will return all counted objects instead of number of objects
std.annotate(Count('s_age')).values('s_name','s_age') 	//it returns all counted objects name and age
std.values('s_age').annotate(Count('s_age'))	//it returns all age and how many students belongs to that age --it will visit each object s_age and get its value and see(count) how many 						that values are there in s_age columns
std.values('department','s_age').annotate(Count('department'),Count('s_age')) 	//it returns how many particular age students are there in each department
